## Examples

```javascript
var fs = require('fs');

var directory_listing = fs.list('.');             // get a list of files
console.log(directory_listing);                   // in the current dir
// .,..,scriptfile.js,testdir
var directory_listing = fs.list('./');            // trailing slash is
console.log(directory_listing);                   // optional
// .,..,scriptfile.js,testdir
var directory_listing = fs.list('./s*');          // no wildcard-search
console.log(directory_listing);                   // or the like. Returns
//                                                // an empty object.

var directory_listing = fs.list('./testdir');     // get a list of a
console.log(directory_listing);                   // subdir
// .,..,somefile,subtestdir
var directory_listing = fs.list('./testdir/subtestdir/'); // the subdir
console.log(directory_listing);                   // of a subdir which
// .,..                                           // is empty

var directory_listing = fs.list('./testdir/');    // iterate over the
for (var i in directory_listing) {                // properties of the
    if (directory_listing.hasOwnProperty(i)) {    // object returned by
        console.log(i + " = " + directory_listing[i] ); // fs.list()
    }                                             // looks like an array
  }
// 0 = .
// 1 = ..
// 2 = somefile
// 3 = subtestdir
var directory_listing = fs.list('.');             // watching at a single
console.log(directory_listing[2]);                // entry.
// scriptfile.js

phantom.exit();
```
